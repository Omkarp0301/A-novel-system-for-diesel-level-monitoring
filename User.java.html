package com.example.edi4;

public class User {
    String name, email;

    public User(String name, String email) {
        this.name = name;
        this.email = email;
    }

    public User(){}

    public String getName() {
        return name;
    }

    public String getEmail_id() {
        return email;
    }

    public void setName(String name) {
        this.name = name;
    }

    public void setEmail(String email) {
        this.email = email;
    }
}
